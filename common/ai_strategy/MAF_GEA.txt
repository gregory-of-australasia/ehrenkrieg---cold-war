# Keep MAF units in Africa
KR_keep_MAF_units_home = {
	reversed = yes

	enable_reverse = {
		MAF = { is_in_faction_with = GER }
	}
	enable = {
		is_in_faction_with = MAF
		NOT = {
			tag = MAF
			has_war_with = NFA
		}
	}
	target_array = GER.faction_members

	abort_when_not_enabled = yes

	ai_strategy = {
		type = dont_defend_ally_borders
		id = "MAF"
		value = 1000
	}
}
