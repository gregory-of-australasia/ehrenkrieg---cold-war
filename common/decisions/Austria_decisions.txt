political_actions = {

	AUS_syndie_threat = {
		icon = generic_prepare_civil_war

		allowed = {
			tag = AUS
		}

		available = {
			OR = {
				GER = { surrender_progress > 0.30 }
				52 = { controller = { is_in_faction_with = INT } } # France has occupied Upper Bavaria
				# to make Austria join in the war if it is still a stalemate by 1942
				if = {
					limit = { is_ai = yes }
					AND = {
						is_actual_major = yes
						date > 1942.1.1
						OR = {
							RUS = {
								has_war_with = GER
								NOT = { surrender_progress > 0.10 }
							}
							SOV = {
								has_war_with = GER
								NOT = { surrender_progress > 0.10 }
							}
						}
						FRA = {
							has_war_with = GER
							NOT = { surrender_progress > 0.10 }
						}
					}
				}
				else = {
					hidden_trigger = { always = no }
				}
			}
			OR = {
				has_completed_focus = AUS_status_quo
				has_completed_focus = AUS_embrace_pluralism
				has_completed_focus = AUS_military_occupation
			}
		}
		visible = {
			is_faction_leader = yes
			is_subject = no
			has_socialist_government = no
			can_declare_war_on = FROM
			FROM = { has_war_with = GER }
			NOT = { has_war_with = GER }
		}

		targets = { INT }
		target_root_trigger = { has_completed_focus = AUS_protector_of_the_balkans }
		target_trigger = {
			can_declare_war_on = FROM
			FROM = { has_war_with = GER }
		}

		cost = 0
		fire_only_once = yes

		ai_will_do = {
			factor = 100
			modifier = {
				factor = 0
				has_war = yes
				OR = {
					has_war_with = HUN
					has_war_with = SER
					has_war_with = ROM
					surrender_progress > 0
					HUN = {
						is_subject_of = AUS
						surrender_progress > 0.05
					}
					CZE = {
						is_subject_of = AUS
						surrender_progress > 0.15
					}
					CRO = {
						is_subject_of = AUS
						surrender_progress > 0.15
					}
				}
			}
		}

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision AUS_syndie_threat target: [From.GetName]"
			FROM = { save_event_target_as = threat_country }
			GER = { country_event = aus.tree.4011 }
		}
	}

	AUS_russian_threat = {
		icon = generic_prepare_civil_war

		allowed = { tag = AUS }

		available = {
			OR = {
				is_neighbor_of = RUS
				if = {
					limit = { has_subject = GAL }
					GAL = { is_neighbor_of = RUS }
				}
				else = {
					hidden_trigger = { always = no }
				}
				POL = { is_in_faction_with = RUS }
				custom_trigger_tooltip = {
					tooltip = AUS_poland_fallen_to_russia
					POL = {
						is_in_faction_with = GER
						has_capitulated = yes
					}
					10 = {
						is_owned_by = POL
						controller = { is_ally_with = RUS }
					}
				}
				custom_trigger_tooltip = {
					tooltip = AUS_russia_penetrated_germany
					5 = {
						is_owned_by = GER
						controller = { is_ally_with = RUS }
					}
				}
			}
			OR = {
				has_completed_focus = AUS_status_quo
				has_completed_focus = AUS_embrace_pluralism
				has_completed_focus = AUS_military_occupation
			}
		}

		visible = {
			is_faction_leader = yes
			is_subject = no
			has_socialist_government = no
			can_declare_war_on = FROM
			NOT = { has_war_with = GER }
			FROM = {
				has_war_with = GER
				NOT = { is_in_faction_with = INT }
			}
		}

		cost = 0
		fire_only_once = yes

		targets = { RUS }
		target_root_trigger = { has_completed_focus = AUS_protector_of_the_balkans }
		target_trigger = {
			FROM = {
				has_war_with = GER
				NOT = { is_in_faction_with = INT }
			}
		}

		ai_will_do = {
			factor = 100
			modifier = {
				factor = 0
				GER = {
					surrender_progress < 0.10
					NOT = { has_war_with = INT }
				}
			}
			modifier = {
				factor = 0
				NOT = {
					any_neighbor_country = { is_ally_with = RUS }
					any_allied_country = {
						any_neighbor_country = { is_ally_with = RUS }
					}
				}
			}
			modifier = {
				factor = 0
				has_war = yes
				OR = {
					has_war_with = HUN
					has_war_with = SER
					has_war_with = ROM
					surrender_progress > 0.01
					HUN = {
						is_subject_of = AUS
						surrender_progress > 0.05
					}
					CZE = {
						is_subject_of = AUS
						surrender_progress > 0.15
					}
					CRO = {
						is_subject_of = AUS
						surrender_progress > 0.15
					}
				}
			}
			modifier = {
				factor = 0
				has_war_with = INT
				count_triggers = {
					amount = 2
					POL = {
						surrender_progress < 0.20
						is_in_faction_with = RUS
					}
					SER = {
						surrender_progress < 0.20
						is_in_faction_with = RUS
					}
					ROM = {
						surrender_progress < 0.20
						is_in_faction_with = RUS
					}
					UKR = {
						surrender_progress < 0.20
						is_in_faction_with = RUS
					}
				}
			}
		}

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision AUS_russian_threat target: [From.GetName]"
			FROM = { save_event_target_as = threat_country }
			GER = { country_event = aus.tree.4011 }
		}
	}

	AUS_reform_faction = {
		icon = generic_political_discourse
		allowed = { tag = AUS }

		available = {
			has_war = no
		}

		visible = {
			has_completed_focus = AUS_protector_of_the_balkans
			is_subject = no
			has_socialist_government = no
			OR = {
				is_in_faction = no
				is_in_faction_with = GER
			}
			has_country_flag = AUS_intervened_in_weltkrieg
			NOT = { has_global_flag = AUS_treaty_of_budapest }
		}

		cost = 0
		fire_only_once = no

		ai_will_do = {
			factor = 100
		}

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision AUS_reform_faction"
			clr_country_flag = AUS_intervened_in_weltkrieg
			if = {
				limit = { is_in_faction_with = GER }
				leave_faction = yes
				GER = { country_event = aus.tree.4015 }
			}
			create_faction = austrian_sphere
			every_country = {
				limit = {
					has_country_flag = AUS_former_faction_member
					has_socialist_government = no
					OR = {
						is_in_faction_with = GER
						is_in_faction = no
					}
					is_subject = no
				}
				country_event = aus.tree.4014
			}
		}
	}
}

AUS_integration_of_germany = {
	AUS_integrate_state = {
		icon = eng_trade_unions_demand

		available = {
			FROM = {
				is_fully_controlled_by = ROOT
				compliance > 40
			}
		}

		visible = {
			FROM = {
				is_owned_by = ROOT
				NOT = { is_core_of = ROOT }
			}
		}

		cost = 50
		days_remove = 200
		cancel_if_not_visible = yes

		state_target = yes
		targets = { 50 52 53 54 780 }
		target_root_trigger = {
			NOT = { country_exists = GER }
		}
		target_trigger = {
			FROM = {
				is_owned_by = ROOT
				NOT = { is_core_of = ROOT }
			}
		}

		ai_will_do = {
			factor = 1000
		}

		remove_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision remove AUS_integrate_state target: [From.GetName]"
			hidden_effect = { remove_state_claim = FROM }
			add_state_core = FROM
		}
	}
}
